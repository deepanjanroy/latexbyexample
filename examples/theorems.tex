\documentclass[11pt,a4paper]{report}
\usepackage{amsthm}

\theoremstyle{plain}
\newtheorem{thm}{Theorem}[chapter] % reset theorem numbering for each chapter
\newtheorem*{thm*}{Theorem} % Theorems without numbering

\newtheoremstyle{dotless}% name of the style to be used
  {\topsep}% measure of space to leave above the theorem. E.g.: 3pt
  {\topsep}% measure of space to leave below the theorem. E.g.: 3pt
  {\itshape}% name of font to use in the body of the theorem
  {0pt}% measure of space to indent
  {\bfseries}% name of head font
  { }% punctuation between head and body
  {5pt plus 1 pt minus 1pt}% space after theorem head; " " = normal interword space
  {}% Manually specify head

\newtheoremstyle{supercustom}
	{\topsep}% measure of space to leave above the theorem. E.g.: 3pt
	{\topsep}% measure of space to leave below the theorem. E.g.: 3pt
	{\itshape}% name of font to use in the body of the theorem
	{0pt}% measure of space to indent
	{\bfseries}% name of head font
	{}% punctuation between head and body
	{5pt plus 1 pt minus 1pt}% space after theorem head; " " = normal interword space
	{\thmname{#1}(This is custom)\thmnumber{ #2} -- \thmnote{#3}}% Manually specify head


\theoremstyle{dotless}
\newtheorem*{superplain}{Theorem}

\theoremstyle{supercustom}
\newtheorem*{customhspec}{Foobar}

\theoremstyle{definition}
\newtheorem{defn}[thm]{Definition} % definition numbers are dependent on theorem numbers
\newtheorem{exmp}[thm]{Example} % same for example numbers
